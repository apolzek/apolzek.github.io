Pacotes e Módulos #

Pacote (Package):

Um pacote em Go é o conceito básico de organização de código. Ele é uma coleção de arquivos Go relacionados que fornecem funcionalidades específicas. Os pacotes são a unidade básica de encapsulamento em Go, ajudando a organizar o código de maneira modular e reutilizável. Um pacote pode consistir em vários arquivos, mas todos esses arquivos devem pertencer ao mesmo diretório.

Módulo (Module):

Um módulo em Go é uma unidade de organização de código em nível mais alto, que lida com a dependência e versionamento do código. Módulos são usados para gerenciar dependências externas e versionar o código de uma maneira que facilita a colaboração e a construção de aplicações distribuídas. Um módulo é geralmente associado a um repositório Git e contém informações sobre as dependências, versões e outras configurações relacionadas ao gerenciamento de dependências.

package main

import "fmt"

func main() {
	// Declarando uma variável do tipo int
	var num int = 42

	// Declarando um ponteiro para int e atribuindo o endereço da variável num
	var ptr1 *int = &num

	// Declarando um ponteiro para ponteiro para int e atribuindo o endereço do ponteiro ptr1
	var ptr2 **int = &ptr1

	// Imprimindo os valores
	fmt.Printf("Valor de num: %d\n", num)
	fmt.Printf("Valor apontado por ptr1: %d\n", *ptr1)
	fmt.Printf("Valor apontado por ptr2: %d\n", **ptr2)

	// Modificando o valor através do ponteiro ptr1
	*ptr1 = 84
	fmt.Printf("Novo valor de num: %d\n", num)

	// Modificando o valor através do ponteiro ptr2
	**ptr2 = 120
	fmt.Printf("Novo valor de num: %d\n", num)
}

package main

import (
	"fmt"
	"time"
)

func main() {

	i := 0
	for i < 10 {
		//time.Sleep(time.Second)
		fmt.Println("Incrementando i")
		i++
	}

	for j := 0; j < 10; j += 2 {
		fmt.Println("Incrementando j em 2")
	}
	//fmt.Println(j) Não funciona, j só existe no escopo do for

	names := [3]string{"joao", "maria", "jesus"}
	for index, value := range names {
		fmt.Printf("%d: %s\n", index, value) // Most
	}

	for _, name := range names {
		fmt.Println(name)
	}

	for index, letter := range "WORD" {
		fmt.Println(index, letter) // ASCII number
	}

	for index, letter := range "WORD" {
		fmt.Println(index, string(letter)) // letras
	}

	user := map[string]string{"username": "apolzek", "password": "12345"}
	for key, value := range user {
		fmt.Println(key, value)
	}

	// Não pode usar range em structs

	for {
		fmt.Println("Infinito")
		time.Sleep(time.Second)
	}
}

Panic e Recover # 

package main

import "fmt"

func recoverExec() {
	if r := recover(); r != nil { // recupera
		fmt.Println("trying to recover")
	}

}

func alunoStatus(n1, n2 float64) bool {
	defer recoverExec()
	media := (n1 + n2) / 2
	if media > 6.0 {
		return true
	} else if media < 6 {
		return false
	}

	panic("media == 6 :/") // Mata execução se não tiver um recover
}

func main() {
	// Panic != Error
	fmt.Println(alunoStatus(6, 7))
	fmt.Println(alunoStatus(6, 6))
	fmt.Println("end !")
}

Função Closure # 

package main

import "fmt"

func closure() func() {
	text := "Closure func internal"

	newFunc := func() {
		fmt.Println(text)
	}

	return newFunc
}

func main() {
	text := "Main func internal"
	fmt.Println(text)
	newFuncMain := closure()
	newFuncMain()
}

Embora o Go não use o termo “closure” diretamente, a capacidade de criar funções anônimas que capturam variáveis locais proporciona um comportamento similar ao de closures em outras linguagens de programação.

Interfaces # 

package main

import (
	"fmt"
	"math"
)

type forma interface {
	area() float64
}

type retangulo struct {
	altura  float64
	largura float64
}

type circulo struct {
	raio float64
}

func escreverArea(f forma) {
	fmt.Println("Area da forma eh:", f.area())
}

func (r retangulo) area() float64 {
	return r.altura * r.largura
}

func (c circulo) area() float64 {
	return math.Pi * math.Pow(c.raio, 2)
}

func main() {
	r := retangulo{10, 15}
	escreverArea(r)

	c := circulo{10}
	escreverArea(c)
}
