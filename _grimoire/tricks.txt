###########################
## Enviar texto via ICMP ##
###########################

# Envio
payload=$(echo -n "Hello" | xxd -p)
echo $payload
len=$(($(echo $payload | wc -c | tr -s " ") + 2))
echo $len
ping -s$len -c4 -p$payload 127.0.0.1

# Recebimento
tshark -l -i lo0 -Y "icmp" -T fields -e data | xxd -r
