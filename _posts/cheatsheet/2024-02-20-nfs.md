## Cheatsheet nfs

```sh
# Instalar os pacotes NFS no servidor
sudo apt-get install nfs-kernel-server

# Instalar os pacotes NFS no cliente (Debian/Ubuntu)
sudo apt-get install nfs-common

# Habilitar e iniciar o serviço NFS no servidor
sudo systemctl enable --now nfs-server

# Verificar o status do serviço NFS
sudo systemctl status nfs-server

# Exportar um diretório (adicionar ao /etc/exports)
echo "/diretorio/compartilhado *(rw,sync,no_root_squash)" | sudo tee -a /etc/exports

# Aplicar as mudanças feitas no /etc/exports sem reiniciar o serviço
sudo exportfs -arv

# Mostrar as exportações atuais do servidor NFS
sudo exportfs -v

# Reiniciar o serviço NFS para aplicar mudanças
sudo systemctl restart nfs-server

# Montar um sistema de arquivos NFS no cliente
sudo mount servidor:/diretorio/compartilhado /ponto/de/montagem

# Montar automaticamente um sistema de arquivos NFS no boot (adicionar ao /etc/fstab)
echo "servidor:/diretorio/compartilhado /ponto/de/montagem nfs defaults 0 0" | sudo tee -a /etc/fstab

# Verificar a conectividade NFS (ping)
showmount -e servidor

# Listar todas as exportações disponíveis do servidor NFS
showmount -e

# Desmontar um sistema de arquivos NFS
sudo umount /ponto/de/montagem

# Verificar estatísticas do NFS
nfsstat

# Forçar uma verificação de sistema de arquivos NFS no cliente
sudo mount -o remount /ponto/de/montagem

# Mostrar a tabela de montagem do cliente NFS
cat /proc/mounts | grep nfs

# Atualizar a tabela de exportação no servidor NFS
sudo exportfs -ra

# Verificar a versão do NFS em uso
nfsstat -m

# Verificar quais clientes estão montando um export
sudo showmount -a

# Desabilitar uma exportação (remover do /etc/exports e aplicar com exportfs -arv)
sudo sed -i '/\/diretorio\/compartilhado/d' /etc/exports && sudo exportfs -arv

# Bloquear acesso NFS de um IP específico (editar /etc/exports e reexportar)
sudo sed -i 's/*(rw,/192.168.1.100(ro,/' /etc/exports && sudo exportfs -arv

# Permitir acesso NFS somente leitura (editar /etc/exports para um diretório)
sudo sed -i 's/rw/ro/' /etc/exports && sudo exportfs -arv

# Alterar as opções de exportação para um diretório (editar /etc/exports)
sudo vim /etc/exports

# Reiniciar o cliente NFS (em caso de problemas de montagem/navegação)
sudo systemctl restart nfs-client.target

# Usar o NFS sobre TCP (editar /etc/fstab para adicionar a opção "tcp")
sudo vim /etc/fstab

# Aumentar o desempenho ajustando o tamanho do r/w (editar /etc/fstab)
sudo vim /etc/fstab

# Forçar a utilização de uma versão específica do NFS (editar /etc/fstab)
sudo vim /etc/fstab

# Verificar todos os pontos de montagem NFS e suas opções
mount | grep nfs
```
