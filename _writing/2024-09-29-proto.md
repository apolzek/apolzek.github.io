**saving your time**: *manifests to deploy kafka and rabbitmq as stateful set in kind (k8s local)*



git clone https://github.com/open-telemetry/opentelemetry-proto.git
cd opentelemetry-proto

cat opentelemetry/proto/logs/v1/logs.proto
cat opentelemetry/proto/metrics/v1/metrics.proto
cat opentelemetry/proto/trace/v1/trace.proto


brew install bufbuild/buf/buf
buf build -o -#format=json | jq '.' > schema.json

brew install protobuf
protoc --version

# Ver o schema completo em formato legível
buf build -o -#format=json | jq '.' > schema.json

# Ou exportar como file descriptor set
buf build -o schema.bin


# Gerar um arquivo descriptor com todas as definições
protoc \
  --descriptor_set_out=otlp.desc \
  --include_imports \
  opentelemetry/proto/**/*.proto

# Ver o conteúdo (não é muito legível assim)
protoc --decode_raw < otlp.desc
